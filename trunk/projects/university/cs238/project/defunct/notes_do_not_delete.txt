Notes for some useful debugging code bits

//Display everything from a my_mysql_query
$result = my_trusted_mysql_query( "SELECT * FROM COMPANY" );
foreach( $result as $thingy )
{
  foreach( $thingy as $key => $value )
  {
    echo "$key" . ": " . "$value" . "<br />";
  }  
}

//Display the contents of $_POST
echo "\$_POST contains:";

{
  echo "$key" . ": " . "$value" . "<br />";
}
